package com.evs.electoralofficer.serviceimpl;

import java.util.ArrayList;
import java.util.List;
import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.stereotype.Service;

import com.evs.electoralofficer.dto.EVSVoterDto;
import com.evs.electoralofficer.exception.NoVoterFoundException;
import com.evs.electoralofficer.model.EVSVoter;
import com.evs.electoralofficer.repository.EVSVoterRepository;
import com.evs.electoralofficer.service.EVSElectoralOfficerApiClient;
import com.evs.electoralofficer.service.EVSElectoralOfficerService;
import com.twilio.Twilio;
import com.twilio.rest.api.v2010.account.Message;
import com.twilio.type.PhoneNumber;

@Service
public class EVSElectoralOfficerServiceImpl implements EVSElectoralOfficerService{
	@Value("${twilio.account_sid}")
    private String accountSid;

    @Value("${twilio.auth_token}")
    private String authToken;

    @Value("${twilio.trial_number}")
    private String twilioPhoneNumber;
	
	@Autowired
	private EVSElectoralOfficerApiClient electoralOfficerApiClient;
	
	@Autowired
	private EVSVoterRepository evsVoterRepository;
	
//	@Autowired
//	private EVSElectoralOfficerTwilioProperties electoralOfficerTwilioProperties;
	
	public void sendSms(String to, String message) {
        Twilio.init(accountSid, authToken);
        Message.creator(new PhoneNumber(to), new PhoneNumber(twilioPhoneNumber), message).create();
    }
	
	@Override
	public ResponseEntity<String> approveOrRejectVoter(String voterId) {
		Optional<EVSVoter> evsVoter = evsVoterRepository.findByVoterId(voterId);
		if (evsVoter.isEmpty()) {
			throw new NoVoterFoundException("No voter found");
		}
		else if (evsVoter.get().getEvsVoterStatus().equals("APPROVED")) {
			return new ResponseEntity<>(String.format("Request is already Approved."),HttpStatus.OK);

		}
		else if((evsVoter.get().getAge()>=18) && (evsVoter.get().getEvsVoterStatus().equals("PENDING")))
		{
			evsVoter.get().setEvsVoterStatus("APPROVED");
			evsVoterRepository.save(evsVoter.get());
//			String voterPhoneNumber = new PhoneNumber(evsVoter.get().getContactNo());
//			System.out.println(voterPhoneNumber);
			
			String voterPhoneNumber = "+91"+evsVoter.get().getContactNo();
//			PhoneNumber adminContactNumber = new PhoneNumber(electoralOfficerTwilioProperties.getTrial_numberString());
//			System.out.println(adminContactNumber);
			String message = "Dear Voter your Voter Request is successfully APPROVED, please check it";
//			Message message = Message.creator(voterPhoneNumber, adminContactNumber, mString).create();
			
			Twilio.init(accountSid, authToken);
	        Message.creator(new PhoneNumber(voterPhoneNumber), new PhoneNumber(twilioPhoneNumber), message).create();
			return new ResponseEntity<>(String.format("Successfully Approved voter request for the voter with id %s", voterId),HttpStatus.OK);
		}
		evsVoter.get().setEvsVoterStatus("DECLINED");
		evsVoterRepository.save(evsVoter.get());
//		PhoneNumber voterPhoneNumber = new PhoneNumber(evsVoter.get().getContactNo());
//		PhoneNumber adminContactNumber = new PhoneNumber(electoralOfficerTwilioProperties.getTrial_numberString());
//		Message message = Message.creator(voterPhoneNumber, adminContactNumber, "Dear Voter your Voter Request is DECLINED, please check it").create();
		String voterPhoneNumber = "+91"+evsVoter.get().getContactNo();
//		PhoneNumber adminContactNumber = new PhoneNumber(electoralOfficerTwilioProperties.getTrial_numberString());
//		System.out.println(adminContactNumber);
		String message = "Dear Voter your Voter Request is DECLINED, please check it";
//		Message message = Message.creator(voterPhoneNumber, adminContactNumber, mString).create();
		Twilio.init(accountSid, authToken);
        Message.creator(new PhoneNumber(voterPhoneNumber), new PhoneNumber(twilioPhoneNumber), message).create();
		return new ResponseEntity<>(String.format("Successfully Declined the voter request for the voter with id %s", voterId),HttpStatus.OK);
		
	}

	@Override
	public List<EVSVoterDto> getAllVoters() {
		List<EVSVoterDto> listOfVotersDtos = new ArrayList<>();
		List<EVSVoter> listOfVoters = evsVoterRepository.findAllByEvsVoterStatus("PENDING");
		listOfVoters.forEach((voter)->{
			EVSVoterDto evsVoterDto = new EVSVoterDto(voter.getId(),
					voter.getVoterId(),
					voter.getUserName(),
					voter.getFirstName(),
					voter.getLastName(),
					voter.getGender(),
					voter.getEmail(),
					voter.getDateOfBirth(),
					voter.getAge(),
					voter.getContactNo(),
					voter.getAddress(),
					voter.getNationality(),
					voter.getEvsVoterStatus(),
					voter.getAadharCard());
			listOfVotersDtos.add(evsVoterDto);
		});
		return listOfVotersDtos;
	}

	@Override
	public ResponseEntity<EVSVoterDto> getVoter(String voterId) {
		Optional<EVSVoter> evsOptional = evsVoterRepository.findByVoterId(voterId);
		if (evsOptional.isEmpty()) {
			throw new NoVoterFoundException("No voter found with the voter id "+ voterId);
		}
		EVSVoterDto evsVoterDto = new EVSVoterDto(evsOptional.get().getId(),
				evsOptional.get().getVoterId(),
				evsOptional.get().getUserName(),
				evsOptional.get().getFirstName(),
				evsOptional.get().getLastName(),
				evsOptional.get().getGender(),
				evsOptional.get().getEmail(),
				evsOptional.get().getDateOfBirth(),
				evsOptional.get().getAge(),
				evsOptional.get().getContactNo(),
				evsOptional.get().getAddress(),
				evsOptional.get().getNationality(),
				evsOptional.get().getEvsVoterStatus(),
				evsOptional.get().getAadharCard());
		
		return new ResponseEntity<>(evsVoterDto,HttpStatus.OK);
	}

	public String checkStatus(String userName) {
		
		return electoralOfficerApiClient.checkStatus(userName);
	}

	
}
