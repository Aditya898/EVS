package com.evs.electoralofficer.controller;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import com.evs.electoralofficer.dto.EVSVoterDto;
import com.evs.electoralofficer.serviceimpl.EVSElectoralOfficerServiceImpl;


@RestController
@RequestMapping(value = "electoralOfficer")
public class EVSElectoralOfficerController {
	@Autowired
	private EVSElectoralOfficerServiceImpl electoralOfficerServiceImpl;

	@PutMapping("approveOrDecline/{voterId}")
	public ResponseEntity<String> approveOrDeclineVoterRequest(@PathVariable String voterId)
	{
		return electoralOfficerServiceImpl.approveOrRejectVoter(voterId);
	}
	@GetMapping("allPendingVoters")
	public List<EVSVoterDto> getAllPendingVoters()
	{
		return electoralOfficerServiceImpl.getAllVoters();
	}
	@GetMapping("getVoter/{voterId}")
	public ResponseEntity<EVSVoterDto> getVoter(@PathVariable String voterId)
	{
		return electoralOfficerServiceImpl.getVoter(voterId);
	}
	@GetMapping("checkStatus/{userName}")
    public String checkStatus(@PathVariable String userName)
    {
    	return electoralOfficerServiceImpl.checkStatus(userName);
    } 
}
